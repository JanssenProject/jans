selenium_login: t2_user
store 0 count
while storedVars['count'] < 1000
echo ${x}
open https://c67.gluu.info/identity
waitForTitle oxAuth - Login
type id=loginForm:username t2_user
type id=loginForm:password test
clickAndWait id=loginForm:loginButton
waitForElementPresent css=div.rf-ntf-sum
assertText css=li[class="infomsg"] Welcome, t2_user!
open https://c67.gluu.info/identity/logout
waitForLocation https://c67.gluu.info/identity/authentication/finishlogout
storeEval storedVars['count']=${count}+1
endWhile
